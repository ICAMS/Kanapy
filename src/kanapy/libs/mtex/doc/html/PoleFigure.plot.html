plot pole figure

Syntax 

plot(pf)
plot(pf{1})         % plot only the first pole figure
plot(pf,'contour')  % contour plot
plot(pf,'contourf') % filled contour plot
plot(pf,'smooth')   % smooth plot
plot(pf,'minmax')   % show min and max
mtexColorbar        % show colorbar
